%% ── SLIDE 3: The Question ─────────────────────────────────────────────────────
\begin{frame}
\badge{01}\quad{\color{Navy}\large\textbf{The Question}}
\vspace{0.5em}\par
{\color{Blue}\rule{\linewidth}{0.8pt}}\par
\vspace{0.6em}

Carbon prices are designed to cut CO$_2$ — but do they also \hi{reduce
toxic co-pollutants}?

\vspace{0.8em}

\begin{tcolorbox}[resultbox]
\textbf{Mechanism:} A carbon price raises the cost of fossil-fuel combustion.
Firms respond by cutting output, switching fuels, or adopting cleaner technology.
Each channel also reduces \emph{co-emitted} toxic substances — mercury,
arsenic, lead, VOCs.
\end{tcolorbox}

\vspace{0.8em}

\begin{columns}[T]
\column{0.46\textwidth}
\textcolor{MidGray}{\small\textit{US evidence (cap-and-trade)}}\\[3pt]
Pham \& Roach (2024): RGGI cut\\
coal-utility toxic releases by \hi{62\%}

\column{0.46\textwidth}
\textcolor{MidGray}{\small\textit{Canadian question (intensity standard)}}\\[3pt]
Does Alberta's SGER/TIER produce\\
\hi{comparable co-abatement}?
\end{columns}
\end{frame}

%% ── SLIDE 7: Empirical Strategy ───────────────────────────────────────────────
\begin{frame}
\badge{05}\quad{\color{Navy}\large\textbf{Empirical Strategy}}
\vspace{0.5em}\par
{\color{Blue}\rule{\linewidth}{0.8pt}}\par
\vspace{0.5em}

\textbf{Two-way fixed effects DiD:}

\begin{align*}
y_{it} &= \beta_1\!\underbrace{(Alberta_i \times SGER_t)}_{\text{Phase 1 effect}}
         + \beta_2\!\underbrace{(Alberta_i \times TIER_t)}_{\text{Phase 2 effect}}\\[4pt]
       &\quad + \mathbf{X}_{it}'\boldsymbol{\gamma}
         + \alpha_i + \delta_t + \varepsilon_{it}
\end{align*}

\vspace{0.3em}

\begin{columns}[T]
\column{0.54\textwidth}
{\small
$Alberta_i = 1$ if facility in Alberta\\[2pt]
$SGER_t = \mathbf{1}[t \geq 2007]$ \quad $TIER_t = \mathbf{1}[t \geq 2020]$\\[2pt]
$\mathbf{X}_{it}$: log GDP, fossil-fuel share, coal/gas price\\[2pt]
$\alpha_i$: facility FE \quad $\delta_t$: year FE\\[2pt]
SE clustered by facility
}

\column{0.42\textwidth}
\begin{tcolorbox}[statbox,title={\small Parallel trends test}]
Event study (ref = 2006)\\[3pt]
Pre-trend joint Wald test:\\
$F(3) = 1.70,\; p = 0.166$\\[3pt]
\grn{$\checkmark$ Fail to reject — consistent}
\end{tcolorbox}
\end{columns}
\end{frame}

%% ── SLIDE 10: Mechanism ───────────────────────────────────────────────────────
\begin{frame}
\badge{07b}\quad{\color{Navy}\large\textbf{Why Are Emissions $\approx$ Releases?}}
\vspace{0.5em}\par
{\color{Blue}\rule{\linewidth}{0.8pt}}\par
\vspace{0.5em}

\begin{columns}[T]
\column{0.50\textwidth}
\textbf{Canadian oil \& gas sector:}

\vspace{0.4em}
\begin{tcolorbox}[resultbox]
\hi{99.2\%} of NPRI releases\\
are to \hi{air}\\[4pt]
{\small Mean air: 384,000 kg/facility-yr\\
Mean non-air: \phantom{0}1,655 kg/facility-yr}\\[4pt]
$\mathrm{cor}(lsumemission, lsumreleases) = 0.999$
\end{tcolorbox}

\column{0.46\textwidth}
\textbf{US coal utilities:}\\[4pt]
{\small Coal ash $\rightarrow$ land disposal\\
Scrubber slurry $\rightarrow$ water\\
So $lsumreleases \gg lsumemission$}

\vspace{0.7em}
\textbf{Implication:}\\[4pt]
{\small In Canada, the mechanism is\\
\hi{output curtailment}, not\\
end-of-pipe or media substitution.\\[4pt]
Fewer barrels/m$^3$ produced\\
$\rightarrow$ less venting/flaring\\
$\rightarrow$ lower toxic air emissions}
\end{columns}
\end{frame}

%% ── SLIDE 11: Takeaway ────────────────────────────────────────────────────────
\begin{frame}
\badge{08}\quad{\color{Navy}\large\textbf{Takeaway}}
\vspace{0.3em}\par
{\color{Blue}\rule{\linewidth}{0.8pt}}\par
\vspace{0.35em}

\begin{itemize}\setlength\itemsep{0.5em}

\item \textbf{Co-abatement exists under intensity pricing.}\\
  {\small SGER still reduced toxic air emissions by \hi{17.5\%} — sufficient
  to shift firm technology.}

\item \textbf{Tighter prices deliver larger gains.}\\
  {\small TIER's higher price added a further \hi{26.5\%}.
  Price level matters, not just instrument type.}

\item \textbf{But cap-and-trade delivers more.}\\
  {\small Canada's \hi{$-39\%$} is half the US RGGI \hi{$-62\%$} benchmark.
  A hard cap removes the output-expansion escape valve.}

\end{itemize}

\vspace{0.3em}
\begin{tcolorbox}[resultbox]
\small As Canada's carbon price escalates toward \$170/tCO$_2$e by 2030,
further co-abatement can be expected — these health co-benefits should
enter policy benefit-cost analyses.
\end{tcolorbox}
\end{frame}
